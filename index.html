      public static int rev(int [] array, int start, int end) {
      int temp;
      if(start >= end) return 0;
      temp = array[start];
      array[start] = array[end];
      array[end] = temp;
      return rev(array, start+1, end-1);}
      }

      public static int binomial(int n, int k){
      if (k > n)return 0;
      if (k == 0 || k == n)return 1;
      return binomial(n - 1, k - 1) + binomial(n - 1, k);}
      }
